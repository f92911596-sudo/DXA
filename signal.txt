local HttpService = game:GetService("HttpService")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- هذا هو رابط GitHub الخاص بك
local SIGNAL_URL = "https://raw.githubusercontent.com/f92911596-sudo/DXA/refs/heads/main/signal.txt"

print("📡 [DSC SYSTEM] نظام الاستقبال من GitHub جاهز ويعمل...")

task.spawn(function()
    while task.wait(2) do 
        local success, response = pcall(function()
            -- جلب محتوى الملف من GitHub
            return game:HttpGet(SIGNAL_URL .. "?nc=" .. tick()) -- الـ tick هنا يضمن جلب التحديث فوراً
        end)
        
        -- إذا نجح الاتصال ووجد كلمة KICK_NOW في الملف
        if success and string.find(response, "KICK_NOW") then
            print("🚨 [DSC SYSTEM] وصلت إشارة الطرد من GitHub! جاري الطرد...")
            
            -- طرد اللاعب
            LocalPlayer:Kick("\n[DSC SYSTEM]\nتم الطرد بأمر النظام!")
            
            -- توقيف السكربت
            break
        end
    end
end)
